    vec4 spec;

struct v2f_img {
    vec4 pos;
    vec2 uv;
};
struct appdata_img {
    vec4 vertex;
    vec2 texcoord;
};
struct v2f {
    vec4 pos;
    vec2 uv;
    vec3 ray;
};
struct appdata {
    vec4 vertex;
    vec2 texcoord;
    vec3 texcoord1;
};

v2f vert( in appdata v );
v2f vert( in appdata v ) {
    v2f o;
    o.pos = ( gl_ModelViewProjectionMatrix * v.vertex );
    o.uv = v.texcoord;
    o.ray = v.texcoord1;
    return o;
}
void main() {
    v2f xl_retval;
    appdata xlt_v;
    xlt_v.vertex = vec4( gl_Vertex);
    xlt_v.texcoord = vec2( gl_MultiTexCoord0);
    xlt_v.texcoord1 = vec3( gl_MultiTexCoord1);
    xl_retval = vert( xlt_v);
    gl_Position = vec4( xl_retval.pos);
    gl_TexCoord[0] = vec4( xl_retval.uv, 0.0, 0.0);
    gl_TexCoord[1] = vec4( xl_retval.ray, 0.0);
}
