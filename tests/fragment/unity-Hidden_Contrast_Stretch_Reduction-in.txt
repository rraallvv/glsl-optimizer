struct v2f {
    vec4 position;
    vec2 uv[4];
};
uniform sampler2D _MainTex;
vec4 frag( in v2f i );
vec4 frag( in v2f i ) {
    vec4 v00;
    vec2 v01;
    vec2 v10;
    vec2 v11;
    vec4 res;
    v00 = texture2D( _MainTex, i.uv[ 0 ]);
    v01 = texture2D( _MainTex, i.uv[ 1 ]).xy ;
    v10 = texture2D( _MainTex, i.uv[ 2 ]).xy ;
    v11 = texture2D( _MainTex, i.uv[ 3 ]).xy ;
    res.x  = max( max( v00.x , v01.x ), max( v10.x , v11.x ));
    res.y  = min( min( v00.y , v01.y ), min( v10.y , v11.y ));
    res.zw  = v00.zw ;
    return res;
}
void main() {
    vec4 xl_retval;
    v2f xlt_i;
    xlt_i.position = vec4(0.0);
    xlt_i.uv[0] = vec2( gl_TexCoord[0]);
    xlt_i.uv[1] = vec2( gl_TexCoord[1]);
    xlt_i.uv[2] = vec2( gl_TexCoord[2]);
    xlt_i.uv[3] = vec2( gl_TexCoord[3]);
    xl_retval = frag( xlt_i);
    gl_FragData[0] = vec4( xl_retval);
}
