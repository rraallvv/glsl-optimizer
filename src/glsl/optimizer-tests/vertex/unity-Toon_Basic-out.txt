struct v2f {
  vec4 pos;
  vec2 texcoord;
  vec3 cubenormal;
};
struct appdata {
  vec4 vertex;
  vec2 texcoord;
  vec3 normal;
};
uniform vec4 _MainTex_ST;
void main ()
{
  appdata xlt_v;
  xlt_v.vertex = gl_Vertex.xyzw;
  xlt_v.texcoord = gl_MultiTexCoord0.xy;
  xlt_v.normal = gl_Normal.xyz;
  v2f o;
  o.pos = (gl_ModelViewProjectionMatrix * xlt_v.vertex);
  o.texcoord = ((xlt_v.texcoord.xy * _MainTex_ST.xy) + _MainTex_ST.zw);
  vec4 tmpvar_18;
  tmpvar_18.xyz = vec3(xlt_v.normal.xyzx);
  tmpvar_18.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  o.cubenormal = (gl_ModelViewMatrix * tmpvar_18).xyz;
  gl_Position = o.pos.xyzw;
  vec4 tmpvar_11;
  tmpvar_11.xy = vec2(o.texcoord.xyxx);
  tmpvar_11.z = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_11.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[0] = tmpvar_11;
  vec4 tmpvar_13;
  tmpvar_13.xyz = vec3(o.cubenormal.xyzx);
  tmpvar_13.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[1] = tmpvar_13;
}
