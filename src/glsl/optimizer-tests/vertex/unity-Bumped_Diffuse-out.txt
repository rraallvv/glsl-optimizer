struct v2f_surf {
  vec4 pos;
  vec2 hip_pack0;
  vec3 TtoV0;
  vec3 TtoV1;
  vec3 TtoV2;
};
struct appdata_full {
  vec4 vertex;
  vec4 tangent;
  vec3 normal;
  vec4 texcoord;
  vec4 texcoord1;
  vec4 color;
};
uniform vec4 _BumpMap_ST;
attribute vec4 TANGENT;
void main ()
{

  xlt_v.texcoord1 = gl_MultiTexCoord1.xyzw;
  xlt_v.color = gl_Color.xyzw;

  v = xlt_v;
  v2f_surf o;
  o.pos = (gl_ModelViewProjectionMatrix * xlt_v.vertex);
  o.hip_pack0 = ((xlt_v.texcoord.xy * _BumpMap_ST.xy) + _BumpMap_ST.zw).xy;

  tmpvar_52 = cross (v.normal, v.tangent.xyz);

  tmpvar_54[1].y = float(tmpvar_53.yyy);
  tmpvar_54[1].z = float(v.normal.yyy);
  tmpvar_54[2].x = float(v.tangent.zzz);
  tmpvar_54[2].y = float(tmpvar_53.zzz);
  tmpvar_54[2].z = float(v.normal.zzz);

  tmpvar_70[1] = gl_ModelViewMatrixInverseTranspose[1].xyz;
  tmpvar_70[2] = gl_ModelViewMatrixInverseTranspose[2].xyz;

  tmpvar_80[1] = gl_ModelViewMatrixInverseTranspose[1].xyz;
  tmpvar_80[2] = gl_ModelViewMatrixInverseTranspose[2].xyz;
  o.TtoV1 = (tmpvar_54 * tmpvar_80[1].xyz);
  mat3 tmpvar_90;
  tmpvar_90[0] = gl_ModelViewMatrixInverseTranspose[0].xyz;
  tmpvar_90[1] = gl_ModelViewMatrixInverseTranspose[1].xyz;
  tmpvar_90[2] = gl_ModelViewMatrixInverseTranspose[2].xyz;
  o.TtoV2 = (tmpvar_54 * tmpvar_90[2].xyz);
  gl_Position = o.pos.xyzw;
  vec4 tmpvar_17;
  tmpvar_17.xy = vec2(o.hip_pack0.xyxx);
  tmpvar_17.z = float(vec4 (0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_17.w = float(vec4 (0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[0] = tmpvar_17;

  tmpvar_19.xyz = vec3(o.TtoV0.xyzx);
  tmpvar_19.w = float(vec4 (0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[1] = tmpvar_19;
  vec4 tmpvar_21;
  tmpvar_21.xyz = vec3(o.TtoV1.xyzx);
  tmpvar_21.w = float(vec4 (0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[2] = tmpvar_21;
  vec4 tmpvar_23;
  tmpvar_23.xyz = vec3(o.TtoV2.xyzx);
  tmpvar_23.w = float(vec4 (0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[3] = tmpvar_23;
}
