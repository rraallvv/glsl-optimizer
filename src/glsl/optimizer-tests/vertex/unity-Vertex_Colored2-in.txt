struct v2f {
    vec4 pos;
    float fog;
    vec4 color;
    vec2 texcoord;
};
struct appdata {
    vec4 vertex;
    vec3 normal;
    vec4 color;
    vec2 texcoord;
};

void PositionFog( in vec4 v, out vec4 pos, out float fog );
v2f vert( in appdata v );
void PositionFog( in vec4 v, out vec4 pos, out float fog ) {
    pos = ( gl_ModelViewProjectionMatrix * v );
    fog = pos.z ;
}
v2f vert( in appdata v ) {
    v2f o;
    PositionFog( v.vertex, o.pos, o.fog);
    o.color.xyz  = ((v.color.xyz  + (v.normal * 0.300000)) * 0.500000);
    o.color.w  = 1.00000;
    o.texcoord = v.texcoord;
    return o;
}
varying vec4 xlv_FOG;
void main() {
    v2f xl_retval;
    appdata xlt_v;
    xlt_v.vertex = vec4( gl_Vertex);
    xlt_v.normal = vec3( gl_Normal);
    xlt_v.color = vec4( gl_Color);
    xlt_v.texcoord = vec2( gl_MultiTexCoord0);
    xl_retval = vert( xlt_v);
    gl_Position = vec4( xl_retval.pos);
    xlv_FOG = vec4( xl_retval.fog, 0.0, 0.0, 0.0);
    gl_FrontColor = vec4( xl_retval.color);
    gl_TexCoord[0] = vec4( xl_retval.texcoord, 0.0, 0.0);
}
