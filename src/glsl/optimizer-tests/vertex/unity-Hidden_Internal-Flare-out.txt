struct v2f {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
};
struct appdata_t {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
};
uniform vec4 _FlareTexture_ST;
void main ()
{
  appdata_t xlt_v;
  xlt_v.vertex = gl_Vertex.xyzw;
  xlt_v.color = gl_Color.xyzw;
  xlt_v.texcoord = gl_MultiTexCoord0.xy;
  v2f o;
  o.vertex = (gl_ModelViewProjectionMatrix * xlt_v.vertex);
  o.color = xlt_v.color;
  o.texcoord = ((xlt_v.texcoord.xy * _FlareTexture_ST.xy) + _FlareTexture_ST.zw);
  gl_Position = o.vertex.xyzw;
  gl_FrontColor = o.color.xyzw;
  vec4 tmpvar_13;
  tmpvar_13.xy = vec2(o.texcoord.xy);
  tmpvar_13.z = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_13.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[0] = tmpvar_13;
}
