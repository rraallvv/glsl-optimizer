struct v2f_combiner {
    vec4 pos;
    vec2 uv;
};
struct appdata_combiner {
    vec4 vertex;
    vec2 texcoord;
};
struct v2f_vertex_lit {
    vec2 uv;
    vec4 diff;
    vec4 spec;
};
struct v2f_img {
    vec4 pos;
    vec2 uv;
};
struct appdata_img {
    vec4 vertex;
    vec2 texcoord;
};

v2f_combiner vertcombiner( in appdata_combiner v );
v2f_combiner vertcombiner( in appdata_combiner v ) {
    v2f_combiner o;
    o.pos = v.vertex;
    o.pos = ( gl_ModelViewProjectionMatrix * o.pos );
    o.uv = v.texcoord;
    return o;
}
void main() {
    v2f_combiner xl_retval;
    appdata_combiner xlt_v;
    xlt_v.vertex = vec4( gl_Vertex);
    xlt_v.texcoord = vec2( gl_MultiTexCoord0);
    xl_retval = vertcombiner( xlt_v);
    gl_Position = vec4( xl_retval.pos);
    gl_TexCoord[0] = vec4( xl_retval.uv, 0.0, 0.0);
}
