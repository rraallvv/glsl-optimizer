struct v2f {
  vec4 vertex;
  vec4 color;
};
struct appdata_t {
  vec4 vertex;
  vec4 color;
};
void main ()
{
  appdata_t xlt_v;
  xlt_v.vertex = gl_Vertex.xyzw;
  xlt_v.color = gl_Color.xyzw;
  v2f o;
  o.vertex = (gl_ModelViewProjectionMatrix * xlt_v.vertex);
  o.color = xlt_v.color;
  gl_Position = o.vertex.xyzw;
  gl_FrontColor = o.color.xyzw;
}
