struct appdata_base {
  vec4 vertex;
  vec3 normal;
  vec4 texcoord;
};
uniform sampler2D _MainTex;
void main ()
{
  appdata_base xlt_v;
  xlt_v.vertex = gl_Vertex.xyzw;
  xlt_v.normal = gl_Normal.xyz;
  xlt_v.texcoord = gl_MultiTexCoord0.xyzw;
  appdata_base v;
  v = xlt_v;
  vec4 tmpvar_16;
  tmpvar_16.xy = vec2(xlt_v.texcoord.xy);
  tmpvar_16.z = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_16.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  vec4 coord;
  coord = tmpvar_16;
  vec4 tmpvar_19;
  tmpvar_19 = texture2DLod (_MainTex, coord.xy, coord.w);
  v.vertex.y = float(vec2((v.vertex.y + (tmpvar_19.x * 2.000000))));
  gl_Position = (gl_ModelViewProjectionMatrix * v.vertex).xyzw;
  vec4 tmpvar_11;
  tmpvar_11.xy = vec2(v.texcoord.xy);
  tmpvar_11.z = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_11.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[0] = tmpvar_11;
}
