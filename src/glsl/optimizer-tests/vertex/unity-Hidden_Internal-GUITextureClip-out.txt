struct v2f {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
  vec2 texgencoord;
};
struct appdata_t {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
};
uniform vec4 _MainTex_ST;
uniform mat4 _GUIClipTextureMatrix;
void main ()
{
  appdata_t xlt_v;
  xlt_v.vertex = gl_Vertex.xyzw;
  xlt_v.color = gl_Color.xyzw;
  xlt_v.texcoord = gl_MultiTexCoord0.xy;
  v2f o;
  o.vertex = (gl_ModelViewProjectionMatrix * xlt_v.vertex);
  o.texgencoord = (_GUIClipTextureMatrix * (gl_ModelViewMatrix * xlt_v.vertex)).xy;
  o.color = xlt_v.color;
  o.texcoord = ((xlt_v.texcoord.xy * _MainTex_ST.xy) + _MainTex_ST.zw);
  gl_Position = o.vertex.xyzw;
  gl_FrontColor = o.color.xyzw;
  vec4 tmpvar_13;
  tmpvar_13.xy = vec2(o.texcoord.xy);
  tmpvar_13.z = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_13.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[0] = tmpvar_13;
  vec4 tmpvar_15;
  tmpvar_15.xy = vec2(o.texgencoord.xy);
  tmpvar_15.z = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  tmpvar_15.w = float(vec4(0.000000, 0.000000, 0.000000, 0.000000));
  gl_TexCoord[1] = tmpvar_15;
}
