struct v2f {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
};
uniform sampler2D _MainTex;
void main ()
{
  v2f xlt_i;
  xlt_i.vertex = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.color = gl_Color.xyzw;
  xlt_i.texcoord = gl_TexCoord[0].xy;
  v2f i;
  i = xlt_i;
  vec4 prev;
  vec4 tmpvar_11;
  tmpvar_11 = texture2D (_MainTex, xlt_i.texcoord);
  vec4 tmpvar_12;
  tmpvar_12 = (i.color * tmpvar_11);
  prev = tmpvar_12;
  prev.xyz = (tmpvar_12.xyz * tmpvar_12.w).xyz;
  gl_FragData[0] = prev.xyzw;
}
