};

{

  xlt_i.fog = xlv_FOG.x;
  xlt_i.color = gl_Color.xyzw;
  xlt_i.texcoord = gl_TexCoord[0].xy;
  xlt_i.normal = gl_TexCoord[1].xyz;

  vec4 tmpvar_14;
