struct v2f {
  vec4 pos;
  float fog;
  vec4 color;
  vec2 uv;
};
varying vec4 xlv_FOG;
uniform sampler2D _MainTex;
void main ()
{
  v2f xlt_xlat_var_input;
  xlt_xlat_var_input.pos = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_xlat_var_input.fog = xlv_FOG.x;
  xlt_xlat_var_input.color = gl_Color.xyzw;
  xlt_xlat_var_input.uv = gl_TexCoord[0].xy;
  v2f xlat_var_input;
  xlat_var_input = xlt_xlat_var_input;
  vec4 col;
  vec4 tmpvar_12;
  tmpvar_12 = texture2D (_MainTex, xlat_var_input.uv);
  col = tmpvar_12;
  col.xyz = (tmpvar_12.xyz * xlat_var_input.color.xyz).xyz;
  float x;
  x = col.w;
  if ((x < 0.000000)) {
    discard;
  };
  gl_FragData[0] = col.xyzw;
}
