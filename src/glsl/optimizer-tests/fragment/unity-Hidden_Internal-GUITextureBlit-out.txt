struct v2f {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
  vec2 texgencoord;
};
uniform sampler2D _MainTex;
uniform sampler2D _GUIClipTexture;
void main ()
{
  v2f xlt_i;
  xlt_i.vertex = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.color = gl_Color.xyzw;
  xlt_i.texcoord = gl_TexCoord[0].xy;
  xlt_i.texgencoord = gl_TexCoord[1].xy;
  v2f i;
  i = xlt_i;
  vec4 col;
  vec4 tmpvar_13;
  tmpvar_13 = texture2D (_MainTex, i.texcoord);
  col.xyz = (tmpvar_13.xyz * i.color.xyz).xyz;
  vec4 tmpvar_15;
  tmpvar_15 = texture2D (_GUIClipTexture, i.texgencoord);
  col.w = float(vec4((i.color.w * tmpvar_15.w)));
  gl_FragData[0] = col.xyzw;
}
