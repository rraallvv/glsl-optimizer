struct v2f_combiner {
    vec4 pos;
    vec2 uv;
};
struct appdata_combiner {
    vec4 vertex;
    vec2 texcoord;
};
struct v2f_vertex_lit {
    vec2 uv;
    vec4 diff;
    vec4 spec;
};
struct v2f_img {
    vec4 pos;
    vec2 uv;
};

    vec4 vertex;

};
uniform vec4 _Color;
uniform sampler2D _RGBSource;
vec4 frag( in v2f_combiner i );
vec4 frag( in v2f_combiner i ) {
    vec4 c;
    c = texture2D( _RGBSource, i.uv);
    c.x  = _Color.w ;
    return c;
}
void main() {
    vec4 xl_retval;
    v2f_combiner xlt_i;
    xlt_i.pos = vec4(0.0);
    xlt_i.uv = vec2( gl_TexCoord[0]);
    xl_retval = frag( xlt_i);
    gl_FragData[0] = vec4( xl_retval);
}
