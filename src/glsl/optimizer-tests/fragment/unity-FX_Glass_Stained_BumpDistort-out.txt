struct v2f {
  vec4 vertex;
  vec4 uvgrab;
  vec2 uvbump;
  vec2 uvmain;
};
uniform sampler2D _MainTex;
uniform vec4 _GrabTexture_TexelSize;
uniform sampler2D _GrabTexture;
uniform sampler2D _BumpMap;
uniform float _BumpAmt;
void main ()
{
  v2f xlt_i;
  xlt_i.vertex = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.uvgrab = gl_TexCoord[0].xyzw;
  xlt_i.uvbump = gl_TexCoord[1].xy;
  xlt_i.uvmain = gl_TexCoord[2].xy;
  v2f i;
  i = xlt_i;
  vec4 normal;
  normal.xy = ((texture2D (_BumpMap, xlt_i.uvbump).wy * 2.000000) - 1.000000).xy;
  normal.z = float(vec3(sqrt (((1.000000 - (normal.x * normal.x)) - (normal.y * normal.y)))));
  i.uvgrab.xy = ((((normal.xy * _BumpAmt) * _GrabTexture_TexelSize.xy) * xlt_i.uvgrab.z) + xlt_i.uvgrab.xy).xy;
  gl_FragData[0] = (texture2DProj (_GrabTexture, i.uvgrab) * texture2D (_MainTex, i.uvmain)).xyzw;
}
