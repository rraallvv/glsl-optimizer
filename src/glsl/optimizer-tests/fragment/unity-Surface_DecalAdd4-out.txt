varying vec4 xlv_FOG;
uniform sampler2D unity_Lightmap;
uniform sampler2D _ShadowMapTexture;
uniform sampler2D _Decal;
void main ()
{
  vec4 c;
  c.xyz = ((texture2D (_Decal, gl_TexCoord[0].xy).xyz * 0.500000) * min ((2.000000 * texture2D (unity_Lightmap, gl_TexCoord[2].xy).xyz), vec3((texture2DProj (_ShadowMapTexture, gl_TexCoord[3].xyzw).x * 2.000000)))).xyz;
  c.w = 0.000000;
  gl_FragData[0] = c.xyzw;
}
