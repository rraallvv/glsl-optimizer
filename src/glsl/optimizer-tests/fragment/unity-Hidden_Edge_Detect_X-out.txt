  vec4 pos;

};

{

  xlt_i.pos = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.uv[0] = gl_TexCoord[0].xy;

  xlt_i.uv[2] = gl_TexCoord[2].xy;

  i = xlt_i;
  vec3 diff;
  vec3 p2;
  vec3 p1;
  vec4 original;
  vec4 tmpvar_13;
  tmpvar_13 = texture2D (_MainTex, i.uv[0]);
  original = tmpvar_13;
  p1 = tmpvar_13.xyz;
  vec4 tmpvar_16;
  tmpvar_16 = texture2D (_MainTex, i.uv[1]);
  p2 = tmpvar_16.xyz;
  vec4 tmpvar_18;
  tmpvar_18 = texture2D (_MainTex, i.uv[2]);
  diff = (((p1 * 2.000000) - p2) - tmpvar_18.xyz);
  float tmpvar_21;
  tmpvar_21 = dot (diff, diff);
  if ((tmpvar_21 >= _Treshold)) {

}
