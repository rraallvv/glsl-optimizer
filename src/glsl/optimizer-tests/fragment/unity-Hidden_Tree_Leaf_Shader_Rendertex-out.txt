struct v2f {
  vec4 pos;
  vec2 uv;
  vec3 color;
};
uniform float _HalfOverCutoff;
uniform sampler2D _MainTex;
void main ()
{
  v2f xlt_i;
  xlt_i.pos = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.uv = gl_TexCoord[0].xy;
  xlt_i.color = gl_TexCoord[1].xyz;
  v2f i;
  i = xlt_i;
  vec4 col;
  vec4 tmpvar_11;
  tmpvar_11 = texture2D (_MainTex, i.uv);
  col = tmpvar_11;
  col.xyz = (tmpvar_11.xyz * (i.color * 2.000000)).xyz;
  col.w = float(vec4((col.w * (2.000000 * _HalfOverCutoff))));
  float x;
  x = (col.w - 1.000000);
  if ((x < 0.000000)) {
    discard;
  };
  gl_FragData[0] = col.xyzw;
}
