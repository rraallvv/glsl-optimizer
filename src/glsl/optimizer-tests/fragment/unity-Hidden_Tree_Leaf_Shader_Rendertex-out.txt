uniform sampler2D _MainTex;
uniform float _HalfOverCutoff;
void main ()
{
  vec4 col;
  vec4 tmpvar_11;
  tmpvar_11 = texture2D (_MainTex, gl_TexCoord[0].xy);
  col = tmpvar_11;
  col.xyz = (tmpvar_11.xyz * (gl_TexCoord[1].xyz * 2.0)).xyz;
  col.w = (vec4((col.w * (2.0 * _HalfOverCutoff)))).w;
  float x;
  x = (col.w - 1.0);
  if ((x < 0.0)) {
    discard;
  };
  gl_FragData[0] = col.xyzw;
}
