struct v2f {
  vec4 vertex;
  vec4 color;
  vec2 texcoord;
};
uniform sampler2D _MainTex;
void main ()
{
  v2f xlt_i;
  xlt_i.vertex = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.color = gl_Color.xyzw;
  xlt_i.texcoord = gl_TexCoord[0].xy;
  v2f i;
  i = xlt_i;
  vec4 col;
  vec4 tmpvar_11;
  tmpvar_11 = texture2D (_MainTex, xlt_i.texcoord);
  col.xyz = ((tmpvar_11.xyz * i.color.xyz) * 2.000000).xyz;
  col.w = float(vec4((i.color.w * tmpvar_11.w)));
  vec4 tmpvar_16;
  tmpvar_16 = mix (vec4(0.500000, 0.500000, 0.500000, 0.500000), col, col.wwww);
  gl_FragData[0] = tmpvar_16.xyzw;
}
