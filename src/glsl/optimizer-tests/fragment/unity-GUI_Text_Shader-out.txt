  vec4 vertex;
  vec2 texcoord;
};

uniform sampler2D _MainTex;
void main ()
{
  v2f xlt_i;
  xlt_i.vertex = vec4(0.000000, 0.000000, 0.000000, 0.000000);
  xlt_i.texcoord = gl_TexCoord[0].xy;
  v2f i;
  i = xlt_i;
  vec4 col;
  col = _Color;
  vec4 tmpvar_10;
  tmpvar_10 = texture2D (_MainTex, i.texcoord);
